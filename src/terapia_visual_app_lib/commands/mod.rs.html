1//! # Comandos Tauri para el Frontend
2//!
3//! Este módulo organiza todos los comandos que el frontend puede invocar,
4//! agrupados por funcionalidad para mantener un código limpio y mantenible.
5//!
6//! ## Estructura
7//!
8//! | Módulo | Propósito |
9//! |--------|-----------|
10//! | [`overlay`] | Comandos para la terapia de overlay (pantalla completa) |
11//! | [`settings`] | Comandos para configuración de la aplicación (idioma, etc.) |
12//!
13//! ## Flujo típico
14//!
15//! 1. El frontend llama a un comando usando `invoke()`.
16//! 2. El comando obtiene el estado de la aplicación (`AppState`).
17//! 3. El comando llama al caso de uso correspondiente.
18//! 4. El comando devuelve el resultado al frontend.
19//!
20//! ## Ejemplo desde el frontend
21//!
22//! ```typescript
23//! import { invoke } from '@tauri-apps/api/core';
24//!
25//! // Iniciar la terapia de overlay
26//! await invoke('cmd_start_overlay', { screenWidth: 1920, screenHeight: 1080 });
27//!
28//! // Cambiar el layout
29//! await invoke('cmd_change_overlay_layout', { newLayout: 'Horizontal', screenWidth: 1920, screenHeight: 1080 });
30//!
31//! // Actualizar el idioma
32//! await invoke('cmd_update_app_settings', { newSettings: { language: 'en' } });
33//! ```
34

43/// Devuelve todos los comandos de tauri empaquetados y listo para inyectar en el builder
44pub fn get_handlers<R: tauri::Runtime>(
45) -> impl Fn(tauri::ipc::Invoke<R>) -> bool + Send + Sync + 'static {
46    tauri::generate_handler![
47        // Comandos de Overlay
48        overlay::cmd_get_overlay_config,
49        overlay::cmd_start_overlay,
50        overlay::cmd_stop_overlay,
51        overlay::cmd_update_overlay_config,
52        overlay::cmd_change_overlay_layout,
53        overlay::cmd_update_overlay_zone_color,
54        overlay::cmd_update_overlay_zone_opacity,
55        overlay::cmd_reset_overlay_config,
56        // Comandos de Lectura
57        reading::cmd_get_reading_config,
58        reading::cmd_start_reading_therapy,
59        reading::cmd_stop_reading_therapy,
60        reading::cmd_update_reading_config,
61        reading::cmd_change_reading_layout,
62        reading::cmd_update_reading_zone_color,
63        reading::cmd_update_reading_zone_opacity,
64        reading::cmd_update_reading_settings,
65        reading::cmd_reset_reading_config,
66        reading::cmd_reading_window_resized,
67        // Comandos Globales / Settings
68        settings::cmd_get_app_settings,
69        settings::cmd_update_app_settings,
70        settings::cmd_exit_app,
71    ]
72}
