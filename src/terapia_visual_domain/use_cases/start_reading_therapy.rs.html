3use crate::{
4    domain::reading_therapy_config::ReadingTherapyConfig,
5    ports::{ReadingWindowError, ReadingWindowPort},
6};
7
8/// Inicia la terapia de lectura en una ventana separada.
9///
10/// Si la ventana ya está activa, se actualiza el contenido.
11pub async fn start_reading_therapy(
12    window: &mut dyn ReadingWindowPort,
13    config: &ReadingTherapyConfig,
14    html_content: &str,
15) -> Result<(), ReadingWindowError> {
16    window.show(config, html_content).await
17}
18
19#[cfg(test)]
20mod tests {
21    use super::*;
22    use crate::use_cases::mocks::MockReadingWindow;
23
24    #[tokio::test]
25    async fn test_start_reading_therapy_ok() {
26        let mut window = MockReadingWindow::default();
27        let config = ReadingTherapyConfig::default();
28        let html = "<p>Hola</p>";
29
30        let result = start_reading_therapy(&mut window, &config, html).await;
31
32        assert!(result.is_ok());
33        assert!(window.is_active());
34        assert!(window.show_called);
35        assert_eq!(window.last_html.unwrap(), html);
36    }
37
38    #[tokio::test]
39    async fn test_start_reading_therapy_fails() {

43        };
44        let config = ReadingTherapyConfig::default();
45
