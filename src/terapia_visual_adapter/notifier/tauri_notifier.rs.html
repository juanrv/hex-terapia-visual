1//! # Adaptador de Notificaciones para Tauri
2//!
3//! Este módulo implementa el puerto [`SystemNotifier`] utilizando las capacidades
4//! nativas de Tauri para mostrar notificaciones del sistema y controlar la
5//! bandeja (system tray).
6//!
7//! # Características
8//!
9//! - **Notificaciones del sistema**: Muestra mensajes al usuario utilizando
10//!   `tauri-plugin-notification`.
11//! - **Control de la bandeja**: Cambia el ícono y el tooltip de la bandeja
12//!   según el estado de la terapia (activa/inactiva).
13//! - **Iconos empaquetados**: Los iconos se incrustan en el binario usando
14//!   `include_bytes!`, lo que evita dependencias de archivos externos.
15//!
16//! # Ejemplo de uso
17//!
18//! ```no_run
19//! use terapia_visual_adapter::notifier::TauriSystemNotifier;
20//! use terapia_visual_domain::ports::SystemNotifier;
21//! use tauri::AppHandle;
22//!
23//! # fn example(app_handle: AppHandle) {
24//! // Crear el notificador con iconos empaquetados
25//! # let icon_inactive = &[];
26//! # let icon_active = &[];
27//! let notifier = TauriSystemNotifier::new(app_handle, icon_inactive, icon_active);
28//!
29//! // Mostrar una notificación
30//! # async fn test(notifier: &TauriSystemNotifier) -> Result<(), Box<dyn std::error::Error>> {
31//! notifier.show_message("Terapia", "Terapia iniciada").await?;
32//! # Ok(())
33//! # }
34//! # }
35//! ```

45/// Adaptador de notificaciones del sistema para Tauri.
46///
47/// Implementa el puerto [`SystemNotifier`] utilizando las APIs nativas de Tauri.
48///
49/// # Campos
50///
51/// * `app_handle` - El handle de la aplicación Tauri, necesario para acceder
52///   a la bandeja y mostrar notificaciones.
53/// * `icon_inactive_bytes` - Los bytes del ícono que se muestra cuando la
54///   terapia está inactiva.
55/// * `icon_active_bytes` - Los bytes del ícono que se muestra cuando la
56///   terapia está activa.
57///
58/// # Ejemplos
59///
60/// ```no_run
61/// use terapia_visual_adapter::notifier::TauriSystemNotifier;
62/// use tauri::AppHandle;
63///
64/// # fn example(app_handle: AppHandle) {
65/// # let icon_inactive = &[];
66/// # let icon_active = &[];
67/// let notifier = TauriSystemNotifier::new(
68///     app_handle,
69///     icon_inactive,
70///     icon_active,
71/// );
72/// # }
73/// ```
74pub struct TauriSystemNotifier {
75    app_handle: AppHandle,
76    icon_inactive_bytes: &'static [u8],
77    icon_active_bytes: &'static [u8],
78}
79
80impl TauriSystemNotifier {
81    /// Crea un nuevo adaptador de notificaciones para Tauri.
82    ///
83    /// # Argumentos
84    ///
85    /// * `app_handle` - El handle de la aplicación Tauri.
86    /// * `icon_inactive_bytes` - Bytes del ícono para el estado inactivo.
87    /// * `icon_active_bytes` - Bytes del ícono para el estado activo.
88    ///
89    /// # Ejemplos
90    ///
91    /// ```no_run
92    /// use terapia_visual_adapter::notifier::TauriSystemNotifier;
93    /// use tauri::AppHandle;
94    ///
95    /// # fn example(app_handle: AppHandle) {
96    /// # let icon_inactive = &[];
97    /// # let icon_active = &[];
98    /// let notifier = TauriSystemNotifier::new(
99    ///     app_handle,
100    ///     icon_inactive,
101    ///     icon_active,
102    /// );
103    /// # }
104    /// ```
105    pub fn new(
106        app_handle: AppHandle,
107        icon_inactive_bytes: &'static [u8],
108        icon_active_bytes: &'static [u8],
109    ) -> Self {
110        Self {

112            icon_inactive_bytes,

114        }
115    }

118#[async_trait]
119impl SystemNotifier for TauriSystemNotifier {
120    /// Muestra una notificación del sistema.
121    ///
122    /// Utiliza `tauri-plugin-notification` para mostrar un mensaje emergente
123    /// al usuario.
124    ///
125    /// # Argumentos
126    ///
127    /// * `title` - Título de la notificación.
128    /// * `message` - Cuerpo del mensaje.
129    ///
130    /// # Errores
131    ///
132    /// Devuelve [`NotifierError::TrayError`] si falla la visualización.
133    ///
134    /// # Ejemplos
135    ///
136    /// ```no_run
137    /// # use terapia_visual_adapter::notifier::TauriSystemNotifier;
138    /// # use terapia_visual_domain::ports::SystemNotifier;
139    /// # async fn example(notifier: &TauriSystemNotifier) -> Result<(), Box<dyn std::error::Error>> {
140    /// notifier.show_message("Éxito", "Configuración guardada").await?;
141    /// # Ok(())
142    /// # }
143    /// ```
144    async fn show_message(&self, title: &str, message: &str) -> Result<(), NotifierError> {
145        // Usar notificación del sistema (Windows toast / Linux notify)
146        self.app_handle
147            .notification()
148            .builder()
149            .title(title)
150            .body(message)
151            .show()
152            .map_err(|e| NotifierError::TrayError(e.to_string()))?;
153
154        info!("Mostrando notificacion: {} - {}", title, message);
155        Ok(())
156    }
157
158    /// Cambia el estado visual de la bandeja (tooltip e ícono).
159    ///
160    /// Esta función actualiza:
161    /// - El tooltip de la bandeja (traducido según el idioma configurado).
162    /// - El ícono de la bandeja (activo/inactivo).
163    ///
164    /// # Argumentos
165    ///
166    /// * `active` - `true` para estado activo, `false` para inactivo.
167    ///
168    /// # Errores
169    ///
170    /// Devuelve [`NotifierError::IconError`] si no se encuentra la bandeja
171    /// o falla la actualización del ícono.
172    ///
173    /// # Ejemplos
174    ///
175    /// ```no_run
176    /// # use terapia_visual_adapter::notifier::TauriSystemNotifier;
177    /// # use terapia_visual_domain::ports::SystemNotifier;
178    /// # async fn example(notifier: &TauriSystemNotifier) -> Result<(), Box<dyn std::error::Error>> {
179    /// // Cambiar a estado activo
180    /// notifier.set_tray_state(true).await?;
181    ///
182    /// // Cambiar a estado inactivo
183    /// notifier.set_tray_state(false).await?;
184    /// # Ok(())
185    /// # }
186    /// ```
187    async fn set_tray_state(&self, active: bool) -> Result<(), NotifierError> {
188        let tray = self.app_handle.tray_by_id("main").ok_or_else(|| {
189            let msg = "Tray icon with id 'main' was not found";
190            warn!("{}", msg);
191            NotifierError::IconError(msg.to_string())
192        })?;
193
194        // Actualizar el tooltip
195        let new_tooltip = if active {
196            messages::tooltip_therapy_active()
197        } else {
198            messages::tooltip_therapy_inactive()
199        };
200
201        tray.set_tooltip(Some(new_tooltip))
202            .map_err(|e| NotifierError::IconError(format!("Error setting tooltip: {}", e)))?;
203
204        // Actualizar icono
205        let bytes = if active {
206            self.icon_active_bytes
207        } else {
208            self.icon_inactive_bytes
209        };
210
211        // Convertir bytes inscrustado a una imagen
212        if let Ok(icon) = tauri::image::Image::from_bytes(bytes) {
213            tray.set_icon(Some(icon))
214                .map_err(|e| NotifierError::IconError(format!("Error setting icon: {}", e)))?;
215        }
216
217        info!("Tray status updated to: {}", active);
218        Ok(())
219    }
220}
